    function initDB() {
        return new Promise((resolve, reject) => {
            const request = indexedDB.open(dbName, 1);
            request.onupgradeneeded = (e) => {
                db = e.target.result;
                if (!db.objectStoreNames.contains(storeName)) {
                    db.createObjectStore(storeName);
                }
            };
            request.onsuccess = (e) => { db = e.target.result; resolve(db); };
            request.onerror = (e) => reject(e);
        });
    }

    function compressImage(base64Str, maxWidth = 800, quality = 0.6) {
        return new Promise((resolve) => {
            const img = new Image();
            img.src = base64Str;
            img.onload = () => {
                let w = img.width; let h = img.height;
                if (w > maxWidth) { h = Math.round((h * maxWidth) / w); w = maxWidth; }
                const canvas = document.createElement('canvas');
                canvas.width = w; canvas.height = h;
                const ctx = canvas.getContext('2d');
                ctx.drawImage(img, 0, 0, w, h);
                resolve(canvas.toDataURL('image/jpeg', quality));
            };
            img.onerror = () => resolve(base64Str);
        });
    }

    function triggerImageUpload(noteId) {
        const input = document.createElement('input'); input.type = 'file'; input.accept = 'image/*'; input.multiple = true;

            files.forEach((file, index) => {
                const reader = new FileReader();
                reader.onload = async (ev) => {
                    const compressedSrc = await compressImage(ev.target.result);
                    const note = notesData.find(n => n.id === noteId);
                    if(note) {
                        note.images = note.images || [];
                        note.images.push({ id: uuid(), src: compressedSrc, x: 20 + offsetBase + (index * 30), y: 20 + offsetBase + (index * 30), w: 200, h: 200 });

                    loadedCount++;
                    if(loadedCount === files.length) {
                        saveDataQuietly();
                        renderImagesToContainer(noteId, `images-${noteId}`);
                        if(activeExpandNoteId === noteId) renderImagesToContainer(noteId, 'expand-images-layer');
                    }
                }; reader.readAsDataURL(file);
            });
        }; input.click();
    }

    function handleNoteDrop(e, noteId) {
        e.preventDefault(); const files = e.dataTransfer.files; if(!files || files.length === 0) return;
        const note = notesData.find(n => n.id === noteId); if(!note) return;
        const rect = e.currentTarget.getBoundingClientRect();
        const dropX = e.clientX - rect.left; const dropY = e.clientY - rect.top;
        const validFiles = Array.from(files).filter(f => f.type.startsWith('image/'));
        if(validFiles.length === 0) return;

        let loadedCount = 0;
        validFiles.forEach((file, index) => {
            const reader = new FileReader();
            reader.onload = async (ev) => {
                const compressedSrc = await compressImage(ev.target.result);
                note.images = note.images || [];
                note.images.push({ id: uuid(), src: compressedSrc, x: Math.max(10, dropX + (index * 30)), y: Math.max(10, dropY + (index * 30)), w: 250, h: 250 });
                loadedCount++;
                if(loadedCount === validFiles.length) {
                    saveDataQuietly(); renderImagesToContainer(noteId, `images-${noteId}`);
                }
            }; reader.readAsDataURL(file);
        });
    }

    function changeMonth(step) { 
        calDate.setMonth(calDate.getMonth() + step); 
        renderCalendar(); 
    }

    function renderCalendar() {
        const year = calDate.getFullYear(); const month = calDate.getMonth(); document.getElementById('cal-month-title').innerText = `${year}년 ${month + 1}월`;
        const firstDay = new Date(year, month, 1).getDay(); const lastDate = new Date(year, month + 1, 0).getDate(); const grid = document.getElementById('cal-grid'); grid.innerHTML = '';
        const today = new Date(); const isCurrentMonth = today.getFullYear() === year && today.getMonth() === month;
        for(let i=0; i<firstDay; i++) grid.innerHTML += `<div class="cal-cell empty"></div>`; 
        for(let i=1; i<=lastDate; i++) {
            const dateStr = `${year}-${String(month+1).padStart(2, '0')}-${String(i).padStart(2, '0')}`;
            const schedule = scheduleData[dateStr]; let recordHtml = '';
            if (schedule && schedule.text) { const doneClass = schedule.done ? 'done' : ''; recordHtml = `<div class="cal-record ${doneClass}">${schedule.text}</div>`; }
            const isTodayClass = (isCurrentMonth && today.getDate() === i) ? 'today' : '';
            const safeText = schedule && schedule.text ? schedule.text.replace(/'/g, "\\'") : ''; const isDone = schedule ? schedule.done : false;
            grid.innerHTML += `<div class="cal-cell ${isTodayClass}" onclick="handleScheduleClick('${dateStr}', '${safeText}', ${isDone})"><span class="cal-date">${i}</span>${recordHtml}</div>`;
        }
    }

    async function handleScheduleClick(dateStr, currentText, isDone) {
        const result = await openModal({ type: 'schedule', title: `📅 ${dateStr}`, message: '일정을 입력하세요.', defaultValue: currentText, isDone: isDone });
        if (result) {
            if (result.action === 'delete' || (result.action === 'save' && result.text === '')) { delete scheduleData[dateStr]; } 
            else if (result.action === 'save') { scheduleData[dateStr] = { text: result.text, done: isDone }; } 
            else if (result.action === 'toggleDone') { scheduleData[dateStr] = { text: result.text, done: !isDone }; }
            saveAndRenderQuietly(); 
        }
    }

    function renderSkills() {
        const list = document.getElementById('skill-list');
        list.innerHTML = '';
        skillsData.forEach(skill => {
            const div = document.createElement('div');
            div.className = 'skill-item';
            div.innerHTML = `
                <div class="skill-name" contenteditable="true" spellcheck="false" onblur="updateSkillName('${skill.id}', this.innerText)">${skill.name}</div>
                <div class="skill-bar-container" onclick="updateSkillProgress(event, '${skill.id}')">
                    <div class="skill-bar-fill" style="width: ${skill.progress}%"></div>
                </div>
                <div class="skill-percent">${skill.progress}%</div>
                <button class="cat-btn del" onclick="deleteSkill('${skill.id}')" style="margin-left:10px;">×</button>
            `;
            list.appendChild(div);
        });
    }

    function updateSkillProgress(e, id) {
        const container = e.currentTarget;
        const rect = container.getBoundingClientRect();
        const clickX = e.clientX - rect.left;
        let percent = Math.round((clickX / rect.width) * 100);
        if(percent < 0) percent = 0; if(percent > 100) percent = 100;
        
        const skill = skillsData.find(s => s.id === id);
        if(skill) { skill.progress = percent; saveDataQuietly(); renderSkills(); }
    }

    function updateSkillName(id, name) {
        const skill = skillsData.find(s => s.id === id);
        if(skill) { skill.name = name; saveDataQuietly(); }
    }

    async function handleAddSkill() {
        const name = await openModal({ type: 'prompt', title: '스킬 추가', message: '연마할 스킬 이름을 적어주세요.', placeholder: '예: 캐릭터 리깅' });
        if(name && name.action === 'save' && name.text) {
            skillsData.push({ id: uuid(), name: name.text, progress: 0 });
            saveDataQuietly(); renderSkills();
        }
    }

    function deleteSkill(id) {
        skillsData = skillsData.filter(s => s.id !== id);
        saveDataQuietly(); renderSkills();
    }

    function triggerGalleryUpload() {
        const input = document.createElement('input'); input.type = 'file'; input.accept = 'image/*'; input.multiple = true;
        input.onchange = (e) => {
            Array.from(e.target.files).forEach(file => {
                const reader = new FileReader();
                reader.onload = async (ev) => {
                    const compressedSrc = await compressImage(ev.target.result, 1200, 0.8);
                    galleryData.push({ id: uuid(), src: compressedSrc });
                    saveDataQuietly(); renderGallery();
                }; reader.readAsDataURL(file);
            });
        }; input.click();
    }

    function deleteGalleryItem(id) {
        galleryData = galleryData.filter(g => g.id !== id);
        saveDataQuietly(); renderGallery();
    }

    async function saveDataQuietly() { 
        if (activeExpandNoteId) {
            const note = notesData.find(n => n.id === activeExpandNoteId);
            if(note) { note.title = document.getElementById('expand-title').innerText; note.content = document.getElementById('expand-body').innerHTML; }
        }
        
        const dataToSave = { categories: categoriesData, notes: notesData, schedule: scheduleData, skills: skillsData, gallery: galleryData };
        if(!db) await initDB();
        const tx = db.transaction(storeName, 'readwrite');
        const store = tx.objectStore(storeName);
        store.put(dataToSave, 'main_save');
    }

    async function initApp() {
        document.execCommand('styleWithCSS', false, true);

        const savedTheme = localStorage.getItem('blender_theme') || 'dark'; 
        document.documentElement.setAttribute('data-theme', savedTheme);
        
        await initDB();
        const tx = db.transaction(storeName, 'readonly');
        const store = tx.objectStore(storeName);
        const request = store.get('main_save');

        request.onsuccess = (e) => {
            let data = e.target.result;
            
            if (!data) {
                const oldRaw = localStorage.getItem('blender_hub_pro_max_v7');
                if (oldRaw) data = JSON.parse(oldRaw);
            }

            if (data) { 
                categoriesData = data.categories || []; notesData = data.notes || []; scheduleData = data.schedule || {}; 
                skillsData = data.skills || skillsData; galleryData = data.gallery || [];
                
                categoriesData.forEach(c => {
                    if(c.isExpanded === undefined) c.isExpanded = true;
                });

            } else { 
                categoriesData = [ {id: uuid(), name: '블렌더 기본 폴더', parentId: null, isExpanded: true} ]; 
            }
            
            saveAndRenderQuietly(); 
            filterItems('all'); 
        };
    }
